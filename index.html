// Google Apps Script - วางใน Apps Script Editor แล้ว Deploy ใหม่ function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let peopleSheet = ss.getSheetByName("People"); let entriesSheet = ss.getSheetByName("Entries"); if (!peopleSheet) { peopleSheet = ss.insertSheet("People"); peopleSheet.appendRow(["id", "name", "color"]); } if (!entriesSheet) { entriesSheet = ss.insertSheet("Entries"); entriesSheet.appendRow(["id", "personId", "day", "start", "end", "location"]); } const action = e && e.parameter && e.parameter.action; if (action === "addPerson") { const id = e.parameter.id; const name = e.parameter.name; const color = e.parameter.color; peopleSheet.appendRow([id, name, color]); return makeResponse({ ok: true }); } if (action === "removePerson") { const id = e.parameter.id; removeRowById(peopleSheet, id); removeRowsByPersonId(entriesSheet, id); return makeResponse({ ok: true }); } if (action === "addEntry") { const id = e.parameter.id; const personId = e.parameter.personId; const day = e.parameter.day; const start = e.parameter.start; const end = e.parameter.end; const location = e.parameter.location || ""; entriesSheet.appendRow([id, personId, day, start, end, location]); return makeResponse({ ok: true }); } if (action === "removeEntry") { const id = e.parameter.id; removeRowById(entriesSheet, id); return makeResponse({ ok: true }); } // Default: return all data const people = sheetToJson(peopleSheet); const entries = sheetToJson(entriesSheet); return makeResponse({ people, entries }); } function doPost(e) { return doGet(e); } function sheetToJson(sheet) { const data = sheet.getDataRange().getValues(); if (data.length < 2) return []; const headers = data[0]; return data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); } function removeRowById(sheet, id) { const data = sheet.getDataRange().getValues(); for (let i = data.length - 1; i >= 1; i--) { if (String(data[i][0]) === String(id)) { sheet.deleteRow(i + 1); } } } function removeRowsByPersonId(sheet, personId) { const data = sheet.getDataRange().getValues(); for (let i = data.length - 1; i >= 1; i--) { if (String(data[i][1]) === String(personId)) { sheet.deleteRow(i + 1); } } } function makeResponse(data) { const output = ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); return output; }
